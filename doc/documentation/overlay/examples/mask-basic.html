Mask

Mask 基本示例。

    1. 全局遮罩Mask组件示例¶

    var Mask = FNX.include('overlay/mask');

var mask = new Mask();

$('#mask-a').click(function() {
    mask.show();
});
$(document).keyup(function(e) {
    // keyboard esc
    if (e.keyCode === 27) {
        mask.hide();
    }
});

    2. 修改遮罩层(绿色、透明度0.3)¶

    var Mask = FNX.include('overlay/mask');

var mask = new Mask();

$('#mask-b').click(function() {
    mask.set('backgroundColor', 'green').set('opacity', '0.3').show();
});
