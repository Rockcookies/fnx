Dialog 示例

Dialog 静态方法示例。

    alert¶

    var Dialog = FNX.include('popup/dialog');

$('#example1').on('click', function () {
    Dialog.alert('信息标题', '信息内容', function () {
        alert('callback')
    });
});

    confirm¶

    var Dialog = FNX.include('popup/dialog');

$('#example2').on('click', function () {
    Dialog.confirm('信息标题', '信息内容', function (flag) {
        alert(flag)
    });
});

    show¶

    var Dialog = FNX.include('popup/dialog');

$('#example3').on('click', function () {
    Dialog.show('信息内容');
});
