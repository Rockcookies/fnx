Dialog 各属性的使用

Dialog 各属性的使用。

    1. complete 事件在iframe中的实践¶

    var Dialog = FNX.include('popup/dialog');

var example1 = new Dialog({
    content: 'iframe:./iframe.html'
});

example1.on('complete', function () {
    console.log('iframe 完全载入成功！');
});

$('#example1').click(function () {
    example1.show();
});

    2. 自定义关闭链接¶

    var Dialog = FNX.include('popup/dialog');

var example2 = new Dialog({
    content: '我是内容啊',
    closeTpl: '点我可以关闭对话框'
});
