    JS¶

    
var Widget = FNX.include('widget/widget');

var WidgetA = Widget.extend({
    events: {
        'click h3': 'heading',
        'mouseover p': 'paragraph'
    },

    heading: function() {
        this.$('h3').html('标题被点击了。');
    },

    paragraph: function() {
        this.$('p').css('background-color', 'red');
    }
});

var a = new WidgetA({ element: '#example1' });
